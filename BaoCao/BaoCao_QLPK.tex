\chapter{PHÂN TÍCH VÀ THIẾT KẾ CƠ SỞ DỮ LIỆU}

\section{Khảo sát nghiệp vụ}

Phòng khám hoạt động theo quy trình: bệnh nhân được lễ tân tiếp nhận và đặt lịch hẹn $\to$ đến ngày hẹn, bác sĩ tạo \emph{Phiếu khám bệnh}, ghi triệu chứng và chẩn đoán $\to$ bác sĩ kê \emph{Toa thuốc} gồm nhiều \emph{Chi tiết đơn thuốc} (số lượng và đơn vị) $\to$ nhân viên thu ngân lập \emph{Hóa đơn} trên cơ sở tiền thuốc (tính từ toa) và tiền khám (lấy từ \emph{Dịch vụ}). Hệ thống lưu lịch sử khám của bệnh nhân, sinh báo cáo doanh thu theo ngày/tháng, báo cáo sử dụng thuốc theo tháng.

CREATE VIEW dbo.v_DanhSachThuoc AS
SELECT  t.maThuoc, t.tenThuoc, t.donGia, t.soLuong,
        t.maDonVi, t.maCachDung,
        dv.tenDonVi, cd.tenCachDung
FROM    Thuoc t
JOIN    DonVi dv     ON t.maDonVi    = dv.maDonVi
JOIN    CachDung cd  ON t.maCachDung = cd.maCachDung;
\end{lstlisting}

\subsection{User-defined Function}

\textbf{(1) Scalar Function} \texttt{f\_TinhTienThuoc} tính tổng tiền thuốc của một Phiếu khám:

\begin{lstlisting}[language=TSQL,caption={Scalar Function tính tiền thuốc}]
CREATE FUNCTION dbo.f_TinhTienThuoc(@maPKB INT)
RETURNS FLOAT
AS
BEGIN
    DECLARE @TongTien FLOAT;
    SELECT  @TongTien = SUM(th.donGia * ctdt.soLuong)
    FROM    ToaThuoc tt
    JOIN    ChiTietDonThuoc ctdt ON tt.maToaThuoc = ctdt.maToaThuoc
    JOIN    Thuoc th             ON ctdt.maThuoc  = th.maThuoc
    WHERE   tt.maPKB = @maPKB;
    RETURN ISNULL(@TongTien, 0);
END;
\end{lstlisting}

\textbf{(2) Table-Valued Function} \texttt{f\_LichSuKhamBenh} trả về lịch sử khám của một bệnh nhân:

\begin{lstlisting}[language=TSQL,caption={Table-Valued Function tra cứu lịch sử khám}]
CREATE FUNCTION dbo.f_LichSuKhamBenh(@maBenhNhan INT)
RETURNS TABLE
AS RETURN
(
    SELECT pkb.maPKB, pkb.ngayKham, pkb.trieuChung,
           STRING_AGG(th.tenThuoc, ', ') AS ToaThuoc,
           (dbo.f_TinhTienThuoc(pkb.maPKB) + ISNULL(hd.tienKham,0)) AS TongChiPhi
    FROM   PhieuKhamBenh pkb
    LEFT JOIN ToaThuoc tt          ON pkb.maPKB     = tt.maPKB
    LEFT JOIN ChiTietDonThuoc ctdt ON tt.maToaThuoc = ctdt.maToaThuoc
    LEFT JOIN Thuoc th             ON ctdt.maThuoc  = th.maThuoc
    LEFT JOIN HoaDon hd            ON pkb.maPKB     = hd.maPKB
    WHERE  pkb.maBenhNhan = @maBenhNhan
    GROUP BY pkb.maPKB, pkb.ngayKham, pkb.trieuChung, hd.tienKham
);
\end{lstlisting}

\subsection{Stored Procedure}

Toàn bộ thao tác Insert/Update/Delete đều đóng gói trong \textbf{25 Stored Procedure} để chống SQL Injection. Các nhóm chính: đăng nhập \& tài khoản, bệnh nhân, bệnh, thuốc, lịch hẹn, phiếu khám (có Transaction), hóa đơn (có Transaction), dịch vụ, báo cáo.

\subsection{Quản lý Giao tác (Transaction)}

Hai nghiệp vụ \emph{Tạo phiếu khám + toa thuốc} và \emph{Lập hóa đơn} được bao trong \texttt{BEGIN TRANSACTION ... COMMIT/ROLLBACK} kèm \texttt{TRY-CATCH}:

\begin{lstlisting}[language=TSQL,caption={SP sp\_LapHoaDonThanhToan -- sử dụng Transaction}]
CREATE PROCEDURE dbo.sp_LapHoaDonThanhToan
    @maPKB INT, @maTaiKhoan INT
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;
    BEGIN TRY
        DECLARE @tienThuoc FLOAT = dbo.f_TinhTienThuoc(@maPKB);
        DECLARE @tienKham FLOAT;
        SELECT @tienKham = tienDichVu FROM DichVu WHERE tenDichVu = N'Khám bệnh';
        IF @tienKham IS NULL SET @tienKham = 50000;

        DECLARE @tongTien FLOAT = @tienThuoc + @tienKham;
        DECLARE @ngayTaiKham DATE;
        SELECT @ngayTaiKham = ngayTaiKham FROM PhieuKhamBenh WHERE maPKB = @maPKB;

        INSERT INTO HoaDon(ngayLapHoaDon, tienThuoc, tienKham, tongTien,
                           ngayTaiKham, maPKB, maTaiKhoan)
        VALUES (GETDATE(), @tienThuoc, @tienKham, @tongTien,
                @ngayTaiKham, @maPKB, @maTaiKhoan);
        COMMIT TRANSACTION;
    END TRY
    BEGIN CATCH
        ROLLBACK TRANSACTION;
        DECLARE @msg NVARCHAR(4000) = ERROR_MESSAGE();
        RAISERROR(@msg, 16, 1);
    END CATCH
END;
\end{lstlisting}

\subsection{Trigger}

Hệ thống cài đặt 4 \textbf{Trigger}:

\begin{enumerate}[leftmargin=*]
\item \texttt{trg\_CapNhatSoLuongThuocTrongKho} -- \texttt{AFTER INSERT}: kiểm tra tồn kho và tự động trừ kho khi kê thuốc.
\item \texttt{trg\_HoanTraSoLuongThuoc} -- \texttt{AFTER DELETE}: hoàn trả tồn kho khi xóa thuốc khỏi đơn.
\item \texttt{trg\_KiemTraNgayDatLich} -- \texttt{AFTER INSERT, UPDATE}: chặn đặt lịch trong quá khứ.
\item \texttt{trg\_ChanXoaTaiKhoan} -- \texttt{INSTEAD OF DELETE}: không cho xóa tài khoản có lịch sử.
\end{enumerate}

\begin{lstlisting}[language=TSQL,caption={Trigger kiểm tra \& cập nhật tồn kho thuốc}]
CREATE TRIGGER trg_CapNhatSoLuongThuocTrongKho
ON ChiTietDonThuoc
AFTER INSERT
AS
BEGIN
    IF EXISTS (SELECT 1 FROM inserted i
               JOIN Thuoc t ON i.maThuoc = t.maThuoc
               WHERE t.soLuong < i.soLuong)
    BEGIN
        RAISERROR(N'Không đủ số lượng thuốc trong kho!', 16, 1);
        ROLLBACK TRANSACTION;
        RETURN;
    END
    UPDATE t SET t.soLuong = t.soLuong - i.soLuong
    FROM Thuoc t JOIN inserted i ON t.maThuoc = i.maThuoc;
END;
\end{lstlisting}

public DataSet LayBaoCaoSuDungThuoc(int thang, int nam)

    using (var cmd = new SqlCommand("sp_BaoCaoSuDungThuoc", con))

protected override void OnModelCreating(ModelBuilder modelBuilder)
{
    // Composite key
    modelBuilder.Entity<ChuanDoan>().HasKey(c => new { c.MaBenh, c.MaPKB });
    modelBuilder.Entity<ChiTietDonThuoc>().HasKey(c => new { c.MaThuoc, c.MaToaThuoc });

    // Unique constraints
    modelBuilder.Entity<BenhNhan>().HasIndex(b => b.CCCD).IsUnique();
    modelBuilder.Entity<TaiKhoan>().HasIndex(t => t.UserName).IsUnique();
    modelBuilder.Entity<Thuoc>().HasIndex(t => t.TenThuoc).IsUnique();

    // CHECK constraints (EF Core 7+ syntax)
    modelBuilder.Entity<Thuoc>()
        .ToTable(t => t.HasCheckConstraint("chk_SoLuongThuoc", "[soLuong] >= 0"));
    modelBuilder.Entity<DichVu>()
        .ToTable(t => t.HasCheckConstraint("chk_TienDichVu",  "[tienDichVu] >= 0"));
    modelBuilder.Entity<ChiTietDonThuoc>()
        .ToTable(t => t.HasCheckConstraint("chk_SoLuongKe",   "[soLuong] > 0"));

    // Default values
    modelBuilder.Entity<LichHen>().Property(l => l.TrangThai)
        .HasDefaultValue("Chờ khám");
    modelBuilder.Entity<PhieuKhamBenh>().Property(p => p.NgayKham)
        .HasDefaultValueSql("(getdate())");

    // Tránh cascade cycle
    modelBuilder.Entity<PhieuKhamBenh>().HasOne(p => p.TaiKhoan).WithMany(t => t.PhieuKhams)
        .HasForeignKey(p => p.MaTaiKhoan).OnDelete(DeleteBehavior.NoAction);
}

    public async Task<ActionResult<IEnumerable<object>>> LichSu(int id)

    public async Task<ActionResult<BenhNhan>> Create([FromBody] BenhNhan dto)
    {
        if (!ModelState.IsValid) return BadRequest(ModelState);
        var created = await _repo.AddAsync(dto);
        return CreatedAtAction(nameof(GetAll), new { id = created.MaBenhNhan }, created);
    }

    [HttpPut("{id:int}")]           // PUT    /api/benhnhan/{id}
    public async Task<IActionResult> Update(int id, [FromBody] BenhNhan dto)
    {
        if (id != dto.MaBenhNhan) return BadRequest("Mã không trùng khớp");
        var ok = await _repo.UpdateAsync(dto);
        return ok ? NoContent() : NotFound();
    }

    [HttpDelete("{id:int}")]        // DELETE /api/benhnhan/{id}
    public async Task<IActionResult> Delete(int id)
        => (await _repo.SoftDeleteAsync(id)) ? NoContent() : NotFound();

\subsection{Tiêu chuẩn RESTful}

Web API tuân thủ các nguyên tắc REST:

\begin{itemize}[leftmargin=*]
\item \textbf{Resource-based URL:} \texttt{/api/benhnhan}, \texttt{/api/hoadon} -- không có verb trong URL.
\item \textbf{HTTP Methods đúng ngữ nghĩa:} \texttt{GET} (đọc), \texttt{POST} (tạo), \texttt{PUT} (cập nhật), \texttt{DELETE} (xóa).
\item \textbf{Status code chuẩn:} \texttt{200 OK}, \texttt{201 Created}, \texttt{204 No Content}, \texttt{400 Bad Request}, \texttt{401 Unauthorized}, \texttt{404 Not Found}.
\item \textbf{Stateless:} mỗi request mang đủ thông tin, không lưu session.
\item \textbf{Content-Type:} \texttt{application/json} (hỗ trợ thêm \texttt{application/xml} qua \texttt{AddXmlSerializerFormatters}).
\end{itemize}

\section{Khó khăn và hướng phát triển}

\subsection{Khó khăn}
\begin{itemize}[leftmargin=*]
\item Đồng bộ schema CSDL giữa các thành viên ban đầu gặp trục trặc; nhóm khắc phục bằng cách tạo \texttt{script.sql} làm chuẩn và đẩy lên Git.
\item Nghiệp vụ kê thuốc -- trừ kho và lập hóa đơn yêu cầu Transaction và Trigger phức tạp, dễ gây lỗi nếu thiếu \texttt{ROLLBACK} trong \texttt{CATCH}.
\item Cú pháp \texttt{HasCheckConstraint} thay đổi giữa EF Core 6 và EF Core 7+, gây lỗi build khi nâng cấp; nhóm khắc phục bằng cách thống nhất dùng EF Core 8 + cú pháp \texttt{ToTable(t => t.HasCheckConstraint(...))}.
\item Mật khẩu hiện đang lưu plaintext trong bảng \texttt{TaiKhoan}; cần băm BCrypt trong giai đoạn nâng cấp bảo mật.
\end{itemize}

\subsection{Hướng phát triển}
\begin{itemize}[leftmargin=*]
\item Bổ sung xác thực \emph{JWT Bearer Token} cho Web API và phân quyền theo Role.
\item Thay thế WinForms client bằng giao diện web (React/Blazor) gọi Web API qua HTTPS.
\item Tích hợp thanh toán điện tử (VNPay/MoMo) và nhắc lịch qua Zalo/SMS.
\item Áp dụng mã hóa mật khẩu (BCrypt) và HTTPS bắt buộc.
\item Mở rộng \texttt{AuditLog} với Trigger ghi mọi thao tác Insert/Update/Delete quan trọng.
\item Triển khai container hóa bằng Docker + CI/CD trên GitHub Actions.
\end{itemize}
